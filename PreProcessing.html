Analisis IRIS di Power BI#

Upload Dataset IRIS ke PostgreSQL#

Mengupload dataset iris ke postgresql dengan nama tabel iris

pip install python-dotenv

import os
from dotenv import load_dotenv
import pandas as pd
from sqlalchemy import create_engine

load_dotenv()
# 1. Load dataset iris dari CSV
df = pd.read_csv("iris.csv")

# 2. Buang baris pertama
df = df.iloc[1:].reset_index(drop=True)

# 3. Rename kolom
df.columns = ["id", "class", "petal_length", "petal_width", "sepal_length", "sepal_width"]

# 4. Koneksi ke PostgreSQL Aiven
host = os.getenv("DB_HOST")
port = os.getenv("DB_PORT")
dbname = os.getenv("DB_NAME")
user = os.getenv("DB_USER")
password = os.getenv("DB_PASSWORD")

# Buat connection string
conn_str = f"postgresql+psycopg2://{user}:{password}@{host}:{port}/{dbname}"
engine = create_engine(conn_str)


# 5. Upload dataset ke tabel iris
df.to_sql("iris", engine, if_exists="replace", index=False)

print("✅ Dataset Iris berhasil diupload ke PostgreSQL tanpa baris pertama!")

from pycaret.clustering import *

# dataset adalah nama tabel input di Power BI

exp_clu = setup(data=dataset)

iris_clusters = assign_model(kmeans_model)
